This guide shows how to implement the GovTech platform on Microsoft Azure. The service interfaces are already defined - you just need to implement the Azure-specific code.

<Note>
**Status:** Structure ready. Implementation time: 3-4 days.

All service interfaces are defined and the Factory pattern is in place. You only need to implement the 4 service classes for Azure.
</Note>

## Azure services mapping

| Platform Service | Azure Service | SDK Package |
|---|---|---|
| StorageService | Blob Storage | `@azure/storage-blob` |
| DatabaseService | Azure Database for PostgreSQL | `pg` (standard PostgreSQL) |
| MonitoringService | Azure Monitor | `@azure/monitor-query` |
| AuthService | Azure Active Directory | `@azure/identity` |
| Kubernetes | Azure Kubernetes Service (AKS) | `@azure/arm-containerservice` |
| Container Registry | Azure Container Registry (ACR) | Built-in |
| Secrets | Azure Key Vault | `@azure/keyvault-secrets` |

## Implementation steps

<Steps>
  <Step title="Install Azure SDKs">
    Install the Azure client libraries:
    
    ```bash npm
    npm install @azure/storage-blob \
      @azure/identity \
      @azure/monitor-query \
      @azure/keyvault-secrets \
      pg
    ```
  </Step>
  
  <Step title="Configure Azure credentials">
    Set up environment variables:
    
    ```env
    CLOUD_PROVIDER=azure
    AZURE_TENANT_ID=your-tenant-id
    AZURE_CLIENT_ID=your-client-id
    AZURE_CLIENT_SECRET=your-client-secret
    AZURE_SUBSCRIPTION_ID=your-subscription-id
    AZURE_STORAGE_ACCOUNT=govtechstorage
    ```
  </Step>
  
  <Step title="Create service principal">
    ```bash
    az ad sp create-for-rbac \
      --name govtech-backend \
      --role Contributor \
      --scopes /subscriptions/{subscription-id}
    ```
    
    Save the output (appId, password, tenant) to your environment variables.
  </Step>
  
  <Step title="Implement service classes">
    Create the 4 service implementation files in `app/backend/src/services/providers/azure/`:
    
    - `azure-storage.service.js`
    - `azure-database.service.js`
    - `azure-monitoring.service.js`
    - `azure-auth.service.js`
  </Step>
  
  <Step title="Update the Factory">
    Uncomment the Azure section in `src/services/factory.js`:
    
    ```javascript
    const PROVIDERS = {
      aws: { /* ... */ },
      azure: {
        storage:    () => new AzureStorageService(),
        database:   () => new AzureDatabaseService(),
        monitoring: () => new AzureMonitoringService(),
        auth:       () => new AzureAuthService(),
      },
    };
    ```
  </Step>
</Steps>

## StorageService implementation

Here's the complete implementation for Azure Blob Storage:

```javascript azure-storage.service.js
// app/backend/src/services/providers/azure/azure-storage.service.js
const { BlobServiceClient } = require('@azure/storage-blob');
const { DefaultAzureCredential } = require('@azure/identity');

class AzureStorageService {
  constructor() {
    const accountName = process.env.AZURE_STORAGE_ACCOUNT;
    const blobServiceClient = new BlobServiceClient(
      `https://${accountName}.blob.core.windows.net`,
      new DefaultAzureCredential()
    );
    this.blobServiceClient = blobServiceClient;
  }

  async uploadFile(containerName, blobName, body, contentType) {
    const containerClient = this.blobServiceClient.getContainerClient(containerName);
    const blockBlobClient = containerClient.getBlockBlobClient(blobName);
    
    await blockBlobClient.upload(body, body.length, {
      blobHTTPHeaders: { blobContentType: contentType },
    });
    
    return { key: blobName, containerName };
  }

  async downloadFile(containerName, blobName) {
    const containerClient = this.blobServiceClient.getContainerClient(containerName);
    const blockBlobClient = containerClient.getBlockBlobClient(blobName);
    
    const downloadResponse = await blockBlobClient.download();
    const chunks = [];
    
    for await (const chunk of downloadResponse.readableStreamBody) {
      chunks.push(chunk);
    }
    
    return Buffer.concat(chunks);
  }

  async deleteFile(containerName, blobName) {
    const containerClient = this.blobServiceClient.getContainerClient(containerName);
    const blockBlobClient = containerClient.getBlockBlobClient(blobName);
    await blockBlobClient.delete();
    return { deleted: true };
  }

  async listFiles(containerName, prefix = '') {
    const containerClient = this.blobServiceClient.getContainerClient(containerName);
    const blobs = [];
    
    for await (const blob of containerClient.listBlobsFlat({ prefix })) {
      blobs.push({
        key: blob.name,
        size: blob.properties.contentLength,
        lastModified: blob.properties.lastModified,
      });
    }
    
    return blobs;
  }

  async getSignedUrl(containerName, blobName, expiresInSeconds = 3600) {
    const containerClient = this.blobServiceClient.getContainerClient(containerName);
    const blockBlobClient = containerClient.getBlockBlobClient(blobName);
    
    const expiryDate = new Date();
    expiryDate.setSeconds(expiryDate.getSeconds() + expiresInSeconds);
    
    const sasUrl = await blockBlobClient.generateSasUrl({
      permissions: 'r',
      expiresOn: expiryDate,
    });
    
    return sasUrl;
  }
}

module.exports = { AzureStorageService };
```

## DatabaseService implementation

```javascript azure-database.service.js
// app/backend/src/services/providers/azure/azure-database.service.js
const { Pool } = require('pg');

class AzureDatabaseService {
  constructor() {
    // Azure Database for PostgreSQL uses standard PostgreSQL protocol
    this.pool = new Pool({
      host: process.env.AZURE_DB_HOST, // e.g., govtech.postgres.database.azure.com
      port: process.env.DB_PORT || 5432,
      database: process.env.DB_NAME,
      user: process.env.DB_USER, // e.g., adminuser@govtech
      password: process.env.DB_PASSWORD,
      ssl: {
        rejectUnauthorized: false, // Azure requires SSL
      },
      max: 20,
      idleTimeoutMillis: 30000
    });
  }

  async query(sql, params = []) {
    const result = await this.pool.query(sql, params);
    return { rows: result.rows, rowCount: result.rowCount };
  }

  async beginTransaction() {
    const client = await this.pool.connect();
    await client.query('BEGIN');
    return client;
  }

  async commitTransaction(client) {
    await client.query('COMMIT');
    client.release();
  }

  async rollbackTransaction(client) {
    await client.query('ROLLBACK');
    client.release();
  }

  async ping() {
    const start = Date.now();
    try {
      await this.pool.query('SELECT 1');
      return { status: 'healthy', latencyMs: Date.now() - start };
    } catch (error) {
      return { status: 'unhealthy', error: error.message };
    }
  }
}

module.exports = { AzureDatabaseService };
```

## AuthService implementation

```javascript azure-auth.service.js
// app/backend/src/services/providers/azure/azure-auth.service.js
const jwt = require('jsonwebtoken');
const { DefaultAzureCredential } = require('@azure/identity');

class AzureAuthService {
  constructor() {
    this.credential = new DefaultAzureCredential();
    this.jwtSecret = process.env.JWT_SECRET || 'your-secret-key';
  }

  async verifyToken(token) {
    try {
      const decoded = jwt.verify(token, this.jwtSecret);
      return { userId: decoded.userId, roles: decoded.roles, valid: true };
    } catch (error) {
      return { valid: false, error: error.message };
    }
  }

  async getUserPermissions(userId) {
    // Implement based on Azure AD roles
    return ['read:documents', 'write:documents'];
  }

  async generateToken(userId, roles = []) {
    return jwt.sign(
      { userId, roles },
      this.jwtSecret,
      { expiresIn: '24h' }
    );
  }

  async revokeToken(token) {
    console.log(`Token revoked: ${token}`);
  }
}

module.exports = { AzureAuthService };
```

## Managed Identity

Use Azure Managed Identity for secure authentication (Azure's equivalent to AWS IRSA):

```bash
# Enable managed identity on AKS
az aks update \
  --resource-group govtech-rg \
  --name govtech-aks \
  --enable-managed-identity

# Create user-assigned managed identity
az identity create \
  --resource-group govtech-rg \
  --name govtech-backend-identity

# Assign permissions
az role assignment create \
  --assignee <identity-client-id> \
  --role "Storage Blob Data Contributor" \
  --scope /subscriptions/{subscription-id}/resourceGroups/govtech-rg
```
